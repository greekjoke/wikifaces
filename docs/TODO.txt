бизнес-партнёры

SELECT DISTINCT ?coOwner ?coOwnerLabel ?companyLabel
WHERE {
  # 1. Identify the target person (e.g., Elon Musk)
  ?person rdfs:label "Elon Musk"@en .
  
  # 2. Find companies owned by the person (P127 = owned by)
  ?company wdt:P127 ?person .
  
  # 3. Find other owners of those same companies
  ?company wdt:P127 ?coOwner .
  
  # 4. Filter to exclude the original person and ensure it's a person (P31=Q5)
  FILTER(?coOwner != ?person)
  ?coOwner wdt:P31 wd:Q5 .                
  
  # 5. Label the results
  SERVICE wikibase:label { bd:serviceParam wikibase:language "[AUTO_LANGUAGE],en". }
}
